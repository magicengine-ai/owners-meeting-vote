# 微信原生鉴权实现指南

## 📚 概述

微信云托管最大的优势：**自动获取微信用户信息，无需维护 access_token**

当用户通过小程序发起请求时，云托管会自动在请求头中注入：
- `X-WX-OPENID`: 用户的 OpenID
- `X-WX-UNIONID`: 用户的 UnionID（如果已绑定）

## 🔧 代码实现

### 1. 创建微信鉴权中间件

创建文件：`backend/src/middleware/wechat_auth.py`

```python
"""
微信原生鉴权中间件
利用云托管自动注入的 X-WX-OPENID 和 X-WX-UNIONID
"""
from fastapi import Request, HTTPException, Depends
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from typing import Optional, Dict
import jwt
from datetime import datetime, timedelta
from ..config import settings


class WeChatAuth:
    """微信鉴权类"""
    
    @staticmethod
    def get_user_info_from_headers(request: Request) -> Dict:
        """
        从请求头获取微信用户信息
        
        Returns:
            dict: {
                "openid": str,      # 用户 OpenID
                "unionid": str,     # 用户 UnionID（可选）
                "is_verified": bool  # 是否已验证
            }
        """
        openid = request.headers.get("X-WX-OPENID")
        unionid = request.headers.get("X-WX-UNIONID")
        
        # 云托管环境：直接使用注入的信息
        if openid:
            return {
                "openid": openid,
                "unionid": unionid,
                "is_verified": True,
                "source": "wechat_cloud"
            }
        
        # 本地开发环境：从 JWT Token 获取
        return WeChatAuth.get_user_from_jwt(request)
    
    @staticmethod
    def get_user_from_jwt(request: Request) -> Dict:
        """
        本地开发环境：从 JWT Token 获取用户信息
        """
        auth_header = request.headers.get("Authorization")
        if not auth_header or not auth_header.startswith("Bearer "):
            return {
                "openid": None,
                "unionid": None,
                "is_verified": False,
                "source": "local_dev"
            }
        
        token = auth_header.split(" ")[1]
        try:
            payload = jwt.decode(
                token,
                settings.JWT_SECRET_KEY,
                algorithms=[settings.JWT_ALGORITHM]
            )
            return {
                "openid": payload.get("sub"),
                "unionid": payload.get("unionid"),
                "is_verified": True,
                "source": "jwt"
            }
        except jwt.InvalidTokenError:
            return {
                "openid": None,
                "unionid": None,
                "is_verified": False,
                "source": "invalid_token"
            }


# 依赖注入：获取当前用户信息
async def get_current_user(request: Request) -> Dict:
    """
    获取当前登录用户信息
    
    用法:
        @app.get("/api/user/profile")
        async def get_profile(current_user: Dict = Depends(get_current_user)):
            return {"openid": current_user["openid"]}
    """
    user_info = WeChatAuth.get_user_info_from_headers(request)
    
    if not user_info["openid"]:
        raise HTTPException(
            status_code=401,
            detail="未登录或登录已过期"
        )
    
    return user_info


# 可选：获取用户信息（不强制登录）
async def get_optional_user(request: Request) -> Dict:
    """
    获取当前用户信息（可选，未登录返回 None）
    
    用法:
        @app.get("/api/public/info")
        async def get_info(current_user: Dict = Depends(get_optional_user)):
            if current_user["openid"]:
                return {"message": f"你好，{current_user['openid']}"}
            return {"message": "游客模式"}
    """
    return WeChatAuth.get_user_info_from_headers(request)
```

```python
from pydantic import BaseModel

class VoteCreateRequest(BaseModel):
    title: str
    options: list[str]
    end_time: datetime

@router.post("/vote/create")
async def create_vote(
    request: VoteCreateRequest,
    current_user: dict = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """
    创建投票
    需要登录且为管理员
    """
    openid = current_user["openid"]
    
    # 验证是否为管理员
    user = db.query(User).filter(User.openid == openid).first()
    if not user or not user.is_admin:
        raise HTTPException(status_code=403, detail="无权限")
    
    # 创建投票
    vote = Vote(
        title=request.title,
        options=request.options,
        end_time=request.end_time,
        creator_openid=openid
    )
    db.add(vote)
    db.commit()
    
    return {"vote_id": vote.id, "status": "created"}
```

#### 示例 3：提交投票（需要业主认证）

```python
class VoteSubmitRequest(BaseModel):
    vote_id: int
    option_index: int

@router.post("/vote/submit")
async def submit_vote(
    request: VoteSubmitRequest,
    current_user: dict = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """
    提交投票
    需要登录且已认证业主身份
    """
    openid = current_user["openid"]
    
    # 验证业主身份
    owner = db.query(Owner).filter(Owner.openid == openid).first()
    if not owner or not owner.is_verified:
        raise HTTPException(
            status_code=403,
            detail="需要完成业主身份认证"
        )
    
    # 验证是否已投票
    existing = db.query(VoteRecord).filter(
        VoteRecord.vote_id == request.vote_id,
        VoteRecord.owner_openid == openid
    ).first()
    
    if existing:
        raise HTTPException(status_code=400, detail="已投票，不能重复投票")
    
    # 提交投票
    vote_record = VoteRecord(
        vote_id=request.vote_id,
        owner_openid=openid,
        option_index=request.option_index
    )
    db.add(vote_record)
    db.commit()
    
    return {"status": "success", "message": "投票成功"}
```

### 3. 免鉴权调用微信接口

创建文件：`backend/src/utils/wechat_api.py`

```python
"""
微信 API 调用工具（免鉴权）
利用云托管的 X-WX-APPID 和 X-WX-SECRET 注入
"""
import httpx
from typing import Optional, Dict, Any
from ..config import settings


class WeChatAPI:
    """微信 API 调用工具类"""
    
    BASE_URL = "https://api.weixin.qq.com"
    
    @classmethod
    async def request(
        cls,
        method: str,
        endpoint: str,
        json_data: Optional[Dict] = None,
        params: Optional[Dict] = None
    ) -> Dict[str, Any]:
        """
        调用微信 API（免鉴权）
        
        云托管会自动注入：
        - X-WX-APPID: 小程序 AppID
        - X-WX-SECRET: 小程序 AppSecret
        
        无需手动维护 access_token！
        """
        url = f"{cls.BASE_URL}{endpoint}"
        
        async with httpx.AsyncClient() as client:
            response = await client.request(
                method=method,
                url=url,
                json=json_data,
                params=params,
                # 云托管会自动添加这些头
                headers={
                    "X-WX-APPID": settings.WECHAT_APP_ID,
                    "X-WX-SECRET": settings.WECHAT_APP_SECRET
                }
            )
            
            result = response.json()
            
            # 检查错误
            if result.get("errcode", 0) != 0:
                raise Exception(f"微信 API 调用失败：{result.get('errmsg')}")
            
            return result
    
    @classmethod
    async def send_subscribe_message(
        cls,
        openid: str,
        template_id: str,
        data: Dict,
        page: str = None
    ) -> Dict:
        """
        发送订阅消息
        
        Args:
            openid: 用户 OpenID
            template_id: 消息模板 ID
            data: 消息数据
            page: 点击消息跳转的小程序页面
        """
        payload = {
            "touser": openid,
            "template_id": template_id,
            "data": data,
        }
        
        if page:
            payload["page"] = page
        
        return await cls.request(
            method="POST",
            endpoint="/cgi-bin/message/subscribe/send",
            json_data=payload
        )
    
    @classmethod
    async def get_phone_number(
        cls,
        code: str
    ) -> Dict:
        """
        获取用户手机号（需要用户授权）
        
        Args:
            code: 小程序 getPhoneNumber 返回的 code
        """
        return await cls.request(
            method="POST",
            endpoint="/wxa/business/getuserphonenumber",
            json_data={"code": code}
        )
```

### 4. 使用示例：发送投票通知

```python
from fastapi import APIRouter, Depends
from src.utils.wechat_api import WeChatAPI
from src.middleware.wechat_auth import get_current_user

router = APIRouter()

@router.post("/notice/send-vote-reminder")
async def send_vote_reminder(
    vote_id: int,
    current_user: dict = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """
    发送投票提醒通知
    """
    openid = current_user["openid"]
    
    # 获取投票信息
    vote = db.query(Vote).filter(Vote.id == vote_id).first()
    if not vote:
        raise HTTPException(status_code=404, detail="投票不存在")
    
    # 发送订阅消息
    await WeChatAPI.send_subscribe_message(
        openid=openid,
        template_id="YOUR_TEMPLATE_ID",  # 在微信公众平台申请
        data={
            "thing1": {"value": vote.title[:20]},
            "time2": {"value": vote.end_time.strftime("%Y-%m-%d %H:%M")},
            "thing3": {"value": "请尽快参与投票"}
        },
        page=f"/pages/vote/detail?id={vote_id}"
    )
    
    return {"status": "success", "message": "通知已发送"}
```
